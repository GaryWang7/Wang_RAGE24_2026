# Whole pipeline:
For step 0-step 2, we run it with shell script to parallelize across libraries. For setp 3, we run it with Docker Rstudio server to combine the individual library results.

## Step 0 - step 2
```sh
# Project/data directories
project_dir=/home/gary/garyw/RAGE24
cellranger_dir=/home/gary/garyw/RAGE24/cellranger_flowcell_combine
out_dir=${project_dir}/CNV/result
tmp_dir=${project_dir}/tmp
mkdir -p $tmp_dir

# Annotation results and reference genome information
anno_csv=${project_dir}/gex_aggr_prep_combine/gex_aggr_anno.csv
ref_dir=${project_dir}/reference/GRCr8

# ref_info_tsv is generated by step0_CNV_prep.R
ref_info_tsv="GCF_036323735.1_GRCr8_assembly_info.tsv"
genome_fa="GCF_036323735.1_GRCr8_mt.fna"

# Bin sizes and whether to assess %N metric for bins
bin_size_Mb=1 # Can use 5Mb bins for easier chromosome-level CNV calling.
bin_N_metrics=FALSE

# CNV calling threshold
frag_library_size_thre=8000 # Filter out cells with fewer than this number of fragments.
#N_percent_thre=30 # Filter out bins with more than this percentage of N bases. Not enabled by default.

# Other resources
barcode_conv_csv=${project_dir}/CNV/resources/barcode_conversion_10Xmultiome.csv
aggcsv=${project_dir}/CNV/resources/CNV_samples.csv
script_dir=/home/gary/garyw/scripts/RAGE/negative_binomial_CNA

# Increase file descriptor limit for parallel jobs
ulimit -n 8192
export TMPDIR=$tmp_dir

# Parallel execution options
PARALLEL_OPTS="--tmpdir $tmp_dir -v --lb --progress --eta --halt soon,fail=1 --jobs 6"

# Step 0
# Create barcode conversion file and CNV sample sheets.

# Step 1--count fragments in each sample.
library_ids=$(cut -f3 -d, $aggcsv | tail -n+2)
cd ${project_dir}

# Can add --dry-run option to test it
parallel $PARALLEL_OPTS --joblog ${tmp_dir}/step1_joblog.txt\
    Rscript ${script_dir}/step1_count_frags.R \
    --library_id {} \
    --cellranger_dir ${cellranger_dir} \
    --out_dir ${out_dir} \
    --bin_size_Mb ${bin_size_Mb} \
    --anno_csv ${anno_csv} \
    --ref_dir ${ref_dir} \
    --ref_info_tsv ${ref_info_tsv} \
    --genome_fasta ${genome_fa} \
    --barcode_conv_csv ${barcode_conv_csv} ::: ${library_ids}

# Step 2
# use library_ids from step 1.
parallel $PARALLEL_OPTS --joblog ${tmp_dir}/step2_joblog.txt\
    Rscript ${script_dir}/step2_chrom_count.R \
    --library_id {} \
    --out_dir ${out_dir} \
    --bin_size_Mb ${bin_size_Mb} \
    --frag_library_size_thre ${frag_library_size_thre} \
    --var_cut_off ${var_cut_off} ::: ${library_ids}

```
